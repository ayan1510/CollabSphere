--- USER SCHEMA (backend/models/User.js) ---
import mongoose from 'mongoose';
import bcrypt from 'bcryptjs';

const userSchema = new mongoose.Schema({
  name: {
    type: String,
    required: true,
    trim: true,
  },
  email: {
    type: String,
    required: true,
    unique: true,
    lowercase: true,
    trim: true,
  },
  password: {
    type: String,
    required: true,
  },
  role: {
    type: String,
    enum: ['employee', 'admin'],
    default: 'employee',
  },
  department: {
    type: String,
    required: true,
    default: 'General',
  },
  skills: {
    type: [String],
    default: [],
  },
  bio: {
    type: String,
    default: '',
  },
  branch: {
    type: String,
    required: true,
    default: 'Headquarters',
  },
  profileImage: {
    type: String,
    default: '',
  },
  points: {
    type: Number,
    default: 0,
  },
  badges: {
    type: [String],
    default: [],
  },
  status: {
    type: String,
    enum: ['active', 'suspended'],
    default: 'active',
  },
}, {
  timestamps: true,
});

// Hash password before saving
userSchema.pre('save', async function (next) {
  if (!this.isModified('password')) {
    return next();
  }
  try {
    const salt = await bcrypt.genSalt(10);
    this.password = await bcrypt.hash(this.password, salt);
    next();
  } catch (error) {
    next(error);
  }
});

// Compare password method
userSchema.methods.comparePassword = async function (enteredPassword) {
  return await bcrypt.compare(enteredPassword, this.password);
};

const User = mongoose.model('User', userSchema);
export default User;

If the server is offline (Network Error / connection refused), fall back to an interactive client-side sandbox powered by localStorage, initialized with seeded mock data. [AI Response Summary]

* Programmed automatic HTTP health pings to the backend `/health` endpoint.
* Created client-side equivalents of all CRUD functions using `localStorage` syncing, guaranteeing seamless presentation if the server is unreachable.
* Added user-friendly indicators, including podium designs for leaderboards, quick login shortcuts, and a pulsing status badge to indicate live sync vs. local sandbox mode.

PHASE 7: MONGODB ATLAS CLOUD MIGRATION
================================================================================
[Engineering Prompt]
"Transition database connectivity from local MongoDB servers to MongoDB Atlas. Solve connection errors indicating authentication failures during server boot."

[AI Response Summary]
* Diagnosed a critical Mongoose connection pattern: if a database name is omitted from the connection string path, Atlas resolves to the `test` database. Since the custom database user was restricted to the `knowledge_sharing` scope, it generated a `bad auth : Authentication failed` error.
* Fixed by updating `MONGODB_URI` to explicitly target `/knowledge_sharing` right before query parameters:
  `mongodb+srv://<username>:<password>@cluster0.r9slwzi.mongodb.net/knowledge_sharing?appName=Cluster0`
* Executed the local database seeder to verify cloud database permissions and pre-seed production data successfully.

PHASE 8: PRODUCTION CORS & ENVIRONMENT SYNCHRONIZATION
================================================================================
[Engineering Prompt]
"Configure CORS dynamically based on environment variables. Split multiple origins from a comma-separated env string and update the React frontend default URL to connect to the live backend URL on Render."

[AI Response Summary]
* Optimized `backend/server.js` using ES Modules syntax to dynamically process CORS requests against a list of origins stored in `process.env.CLIENT_URL`:
  ```javascript
  const allowedOrigins = (process.env.CLIENT_URL || "http://localhost:5173").split(",");
  app.use(cors({ origin: allowedOrigins, credentials: true }));
  ```
* Set Render's backend environment variables to map the production client:
  `CLIENT_URL=http://localhost:5173,https://collab-sphere-delta.vercel.app`
* Adjusted `frontend/src/App.jsx` to direct production requests to the deployed Render backend `https://collabsphere-2-bs1e.onrender.com/api` by default, while allowing overrides via local `frontend/.env` variables.
